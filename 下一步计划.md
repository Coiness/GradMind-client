**代码模板**:
```typescript
import * as math from 'mathjs';

compute: async (inputs, params) => {
  const nComponents = Number(params.nComponents) || 2;

  // 从数据集中提取数据
  const datasetInput = inputs.dataset;
  const data = datasetInput.data || datasetInput;

  // 转换为矩阵
  const X = math.matrix(data);

  // 中心化数据
  const mean = math.mean(X, 0);
  const centered = math.subtract(X, mean);

  // 计算协方差矩阵
  const cov = math.multiply(math.transpose(centered), centered);
  const n = data.length;
  const covMatrix = math.divide(cov, n - 1);

  // 计算特征值和特征向量
  const { values, vectors } = math.eigs(covMatrix);

  // 按特征值排序
  const sorted = values
    .map((val, idx) => ({ value: val, vector: vectors[idx] }))
    .sort((a, b) => b.value - a.value)
    .slice(0, nComponents);

  // 转换数据
  const components = sorted.map(s => s.vector);
  const transformed = math.multiply(centered, math.transpose(components));

  return {
    components: components,
    variance: sorted.map(s => s.value),
    transformed: transformed.toArray(),
    explainedVariance: sorted.map(s => s.value / math.sum(values))
  };
}
```

```typescript
// src/pages/Orchestration/utils/dataTransformer.ts
export function transformData(
  data: any,
  sourceType: string,
  targetType: string
): any {
  // Dataset → Matrix
  if (sourceType === 'dataset' && targetType === 'matrix') {
    if (data.data) return data.data;
    return data;
  }

  // Matrix → Vector（展平）
  if (sourceType === 'matrix' && targetType === 'vector') {
    return data.flat();
  }

  // Vector → Scalar（第一个元素）
  if (sourceType === 'vector' && targetType === 'scalar') {
    return data[0];
  }

  // 不需要转换
  return data;
}
```

```typescript
// 添加到 workflowValidator.ts
function validateDataTypeCompatibility(
  workflow: Workflow,
  algorithmLibrary: AlgorithmNode[]
): ValidationError[] {
  const errors: ValidationError[] = [];

  // 类型兼容性规则
  const compatible: Record<string, string[]> = {
    'dataset': ['matrix', 'vector'],
    'matrix': ['vector', 'matrix'],
    'vector': ['scalar', 'vector'],
    'scalar': ['scalar']
  };

  workflow.edges.forEach(edge => {
    const sourceNode = workflow.nodes.find(n => n.id === edge.source);
    const targetNode = workflow.nodes.find(n => n.id === edge.target);

    const sourceAlg = algorithmLibrary.find(a => a.key === sourceNode?.data.algorithmKey);
    const targetAlg = algorithmLibrary.find(a => a.key === targetNode?.data.algorithmKey);

    if (!sourceAlg || !targetAlg) return;

    const outputPort = sourceAlg.outputs.find(o => o.id === edge.sourceHandle);
    const inputPort = targetAlg.inputs.find(i => i.id === edge.targetHandle);

    if (!outputPort || !inputPort) {
      errors.push({
        edgeId: edge.id,
        message: `无效的端口连接`,
        type: 'error'
      });
      return;
    }

    // 检查类型兼容性
    const sourceType = outputPort.dataType;
    const targetType = inputPort.dataType;

    if (!compatible[sourceType]?.includes(targetType) && sourceType !== targetType) {
      errors.push({
        edgeId: edge.id,
        message: `类型不匹配：${sourceType} 无法连接到 ${targetType}`,
        type: 'error'
      });
    }
  });

  return errors;
}

// 添加到 validateWorkflow 函数
export function validateWorkflow(
  workflow: Workflow,
  algorithmLibrary: AlgorithmNode[]
): ValidationError[] {
  const errors: ValidationError[] = [];

  // 添加类型验证

  return errors;
}
```
